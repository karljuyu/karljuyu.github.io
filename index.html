function doPost(e) {
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getActiveSheet();
  var d = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Timestamp","Full Name","PIN",
      "RR","RR Score","SpO2","SpO2 Sc",
      "SBP","SBP Sc","Pulse","Pulse Sc",
      "Temp","Temp Sc","ACVPU","ACVPU Sc",
      "O2 Supp","O2 Sc","NEWS2 Total","Risk Level"
    ]);
  }
  sheet.appendRow([
    d.timestamp, d.fullName, d.pin,
    d.respRate, d.respScore,
    d.spo2, d.spo2Score,
    d.systolicBP, d.bpScore,
    d.pulse, d.pulseScore,
    d.temp, d.tempScore,
    d.consciousness, d.consciousnessScore,
    d.o2, d.o2Score,
    d.news2Total, d.riskLevel
  ]);
  return ContentService.createTextOutput("OK");
}
